
#include <stdio.h>
#include <string.h>
#include <stdlib.h>


char* strings_xor(char* s, char* t) {
    int n = strlen(s);
    
    char* result = (char*)malloc((n + 1) * sizeof(char));
    for (int i =0; i < n; i++){
        result[i] = (s[i] == t[i]) ? '0' : '1';
    }
    result[n] = '\0';
    return result;
}

int main() {
    char s[100], t[100];
    scanf("%s %s", s, t);
    char* result = strings_xor(s, t);
    
    printf("%s\n", result);
    free(result);
    return 0;
    
}
