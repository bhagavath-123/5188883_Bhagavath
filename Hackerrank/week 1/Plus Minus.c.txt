

#include <stdio.h>

int main ()
{
    int n;
    scanf( " %d", &n );
    int array[n];
    int negative = 0, null = 0, positive = 0;

    for ( int i = 0; i < n; i++ )
    {
        scanf( " %d", &array[i] );
        if ( array[i] < 0 )
            negative++;
        if ( array[i] == 0 )
            null++;
        if ( array[i] > 0 )
            positive++;
    }

    printf( "%f\n%f\n%f\n", (float)positive/n, (float)negative/n, (float)null/n );


    return 0;
